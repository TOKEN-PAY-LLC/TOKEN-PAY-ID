JavaScript — интеграция
Минимальный JS-код для открытия OAuth-попапа и получения токена.
// Вставьте свой CLIENT_ID из личного кабинета TOKEN PAY ID const TPID_CLIENT_ID = 'ваш_client_id'; const TPID_REDIRECT_URI = 'https://yoursite.com/callback'; function loginWithTPID() { const state = crypto.randomUUID(); sessionStorage.setItem('tpid_state', state); const params = new URLSearchParams({ client_id: TPID_CLIENT_ID, redirect_uri: TPID_REDIRECT_URI, response_type: 'code', scope: 'profile email', state, }); const url = `https://tokenpay.space/oauth/authorize?${params}`; const popup = window.open(url, 'tpid_login', 'width=480,height=600,left=' + (screen.width/2-240) + ',top=' + (screen.height/2-300)); const onMsg = async (e) => { if (e.origin !== 'https://tokenpay.space') return; if (e.data?.type !== 'tpid_callback') return; window.removeEventListener('message', onMsg); popup?.close(); const { code, state: retState } = e.data.payload; if (retState !== sessionStorage.getItem('tpid_state')) return; // Обменяйте code → токен на бэкенде const token = await exchangeCode(code); onLoginSuccess(token); }; window.addEventListener('message', onMsg); } async function exchangeCode(code) { const res = await fetch('/api/auth/tpid-callback', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ code, redirect_uri: TPID_REDIRECT_URI }) }); return (await res.json()).access_token; } function onLoginSuccess(token) { console.log('Logged in!', token); // Ваш код после успешного входа }
Backend — обмен code на токен (Node.js)
Обмен authorization code на access token должен происходить на вашем бэкенде для защиты CLIENT_SECRET.
// Express.js endpoint — /api/auth/tpid-callback app.post('/api/auth/tpid-callback', async (req, res) => { const { code, redirect_uri } = req.body; // 1. Обменяем code → токен const tokenRes = await fetch('https://tokenpay.space/api/v1/oauth/token', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ grant_type: 'authorization_code', code, redirect_uri, client_id: process.env.TPID_CLIENT_ID, client_secret: process.env.TPID_CLIENT_SECRET, }) }); const { access_token } = await tokenRes.json(); // 2. Получим профиль пользователя const userRes = await fetch('https://tokenpay.space/api/v1/oauth/userinfo', { headers: { 'Authorization': `Bearer ${access_token}` } }); const user = await userRes.json(); // user = { id, email, name, email_verified, ... } // 3. Найдите/создайте пользователя в вашей БД и верните сессию res.json({ access_token, user }); });
Быстрый старт
Четыре шага для полноценной интеграции.
1
Создайте приложение в личном кабинете
Войдите на tokenpay.space/dashboard → Интеграция → Создать приложение. Вы получите CLIENT_ID и CLIENT_SECRET.
2
Укажите Redirect URI
В настройках приложения добавьте Callback URL: например https://yoursite.com/auth/callback. Без этого авторизация работать не будет.
3
Вставьте кнопку и JS-код
Скопируйте CSS-стили и HTML-кнопку выше. Добавьте функцию loginWithTPID() и замените ваш_client_id на реальный CLIENT_ID.
4
Реализуйте обмен code на backend
Создайте endpoint на вашем сервере, который принимает code и обменивает его на access_token через API TOKEN PAY ID. CLIENT_SECRET никогда не должен быть во фронтенде.
Доступные scope
Запрашивайте только нужные данные пользователя.
profile email openid phone 2fa_status
Пример ответа userinfo: { "id": "tpid_usr_...", "email": "user@example.com", "name": "Иван", "email_verified": true, "phone": "+79001234567" }
